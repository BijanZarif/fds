from scipy.integrate import odeint

def run(u, mu, nsteps):
    x = empty(nsteps)
    for i in range(nsteps):
        u = odeint(lambda u,t : dudt(u, mu), u, [0, 0.01])[1]
        x[i] = u[0]
    return u, x
